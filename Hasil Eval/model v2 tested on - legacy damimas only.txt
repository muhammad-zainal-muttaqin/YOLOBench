from ultralytics import YOLO
import yaml
import os

# ============ DATASET ============
dataset_root = "/kaggle/input/datasets/mzainalmuttaqin2/yolo-damimas-full/yolo_damimas_full_fixed"

data_dict = {
    "path": dataset_root,
    "train": "images/test",
    "val": "images/test",
    "test": "images/test",
    "nc": 4,
    "names": {0: "B1", 1: "B2", 2: "B3", 3: "B4"}
}

yaml_path = "/kaggle/working/data_fixed.yaml"
with open(yaml_path, "w") as f:
    yaml.dump(data_dict, f)
print("YAML saved:", yaml_path)

# ============ MODELS ============
model_base = "/kaggle/input/models/mzainalmuttaqin2/model-test-4th-march-2026/tensorflow2/default/1/Model-4th March 2026"

datasets = ["combined", "damimas", "lonsum"]
model_files = ["y26l_123.pt", "y26l_42.pt", "yv9c_123.pt", "yv9c_42.pt"]

results = []

for ds in datasets:
    for mf in model_files:
        model_path = os.path.join(model_base, ds, mf)

        if not os.path.exists(model_path):
            print(f"[SKIP] Tidak ditemukan: {model_path}")
            continue

        model_name = f"{ds}/{mf}"
        print(f"\n{'='*60}")
        print(f"Evaluating: {model_name}")
        print(f"{'='*60}")

        model = YOLO(model_path)
        metrics = model.val(
            data=yaml_path,
            split="test",
            imgsz=640,
            batch=16,
            conf=0.001,
            iou=0.6,
            device=0
        )

        row = {
            "Dataset": ds,
            "Model": mf.replace(".pt", ""),
            "mAP50": round(metrics.box.map50, 4),
            "mAP50-95": round(metrics.box.map, 4),
            "Precision": round(metrics.box.mp, 4),
            "Recall": round(metrics.box.mr, 4),
        }
        results.append(row)

# ============ SUMMARY TABLE ============
print("\n" + "="*80)
print(" SUMMARY HASIL TEST SEMUA MODEL")
print("="*80)
print(f"{'Dataset':<12} {'Model':<12} {'mAP50':>8} {'mAP50-95':>10} {'Precision':>10} {'Recall':>8}")
print("-"*80)

for r in results:
    print(f"{r['Dataset']:<12} {r['Model']:<12} {r['mAP50']:>8.4f} {r['mAP50-95']:>10.4f} {r['Precision']:>10.4f} {r['Recall']:>8.4f}")

print("-"*80)
